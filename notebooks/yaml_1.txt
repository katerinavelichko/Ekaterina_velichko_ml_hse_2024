services:

    container_name: frontend
    ports:
      - '3000:3000'
    environment:
      NODE_ENV: production
    restart: always
  backend:
    image: gorisso/backend_crypto:1.0.0
    container_name: backend
    ports:
      - '7000:7000'
    restart: always
  bot:
    image: gorisso/bot_crypto_crypto:1.0.0
    container_name: bot
    restart: always
  redis:
    image: redis
    container_name: redis
    ports:
      - '6379:6379'
    restart: always

    image: nginx
    ports:
      - '80:80'
    volumes:
      - ./nginx.conf:/etc/nginx/nginx.conf:ro
